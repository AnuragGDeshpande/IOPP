.hover-effect:hover {

background-color: lightgray;

}

<script>

function createDoc(option) {

if (option >= 1 && option <= 4) {

switch (option) {

case 1:

// Create Google Docs - My Docs

window.open('https://docs.google.com/document/d/14SErYSi2tKDyaXFClqXhVAVEQn8UPsTITtoFq80cgKg/edit?usp=sharing');

break;

case 2:

// Create Google Sheets - My Sheets

window.open('https://docs.google.com/spreadsheets/d/1j23ZDeiZTQmP3vLaQXlXMyhiCWL2GhAuS3YnHE7GSxM/edit?usp=sharing');

break;

case 3:

// Create Google Blog - My Blog

window.open('https://anuraggdeshpandesfirstblog.blogspot.com/2023/06/blog-post.html');

break;

case 4:

// Create Google Slides - My Presentation

window.open('https://docs.google.com/presentation/d/1b7LAGj2qmkNS2FkGVUwGG0MSxW-MZXvWoDcs28dr_lA/edit?usp=sharing');

break;

}

}


function openDocument() {

var input = document.getElementById('docInput').value;

createDoc(parseInt(input));

}

</script>

</head>

<body>

<h1>My Documents</h1>

<ul>

<li class="hover-effect" onclick="createDoc(1)">My Docs</li>

<li class="hover-effect" onclick="createDoc(2)">My Sheets</li>

<li class="hover-effect" onclick="createDoc(3)">My Blog</li>

<li class="hover-effect" onclick="createDoc(4)">My Presentation</li>

</ul>


<label for="docInput">Enter a number (1-4):</label>

<input type="number" id="docInput" min="1" max="4" />

<button onclick="openDocument()">Open Document</button>
